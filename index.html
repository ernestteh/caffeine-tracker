First-time setup

Connect your Google Sheet

Your data is stored in your own private Google Sheet. You need to set one up once — it takes about 5 minutes.

Setup steps:
1. Go to sheets.google.com → create a blank sheet
2. Add these headers in Row 1: Date  Time  Type  Brand  Flavour  Size_ml  Caffeine_per_100  Caffeine_total  Feeling
3. Go to Extensions → Apps Script, paste the script below, save & deploy as a Web App (Execute as: Me, Access: Anyone)
4. Copy the Web App URL and paste it below
Show Apps Script code 
const SHEET_NAME = 'Sheet1';

function doGet(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const rows = data.slice(1)
    .filter(r => r[0] !== '')
    .map(row => {
      const obj = {};
      headers.forEach((h, i) => obj[h] = row[i]);
      return obj;
    });
  return ContentService
    .createTextOutput(JSON.stringify({ ok: true, entries: rows }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME);
  const body = JSON.parse(e.postData.contents);
  if (body.action === 'save_all') {
    const lastRow = sheet.getLastRow();
    if (lastRow > 1) sheet.deleteRows(2, lastRow - 1);
    if (body.entries && body.entries.length > 0) {
      const rows = body.entries.map(entry => [
        entry.Date, entry.Time, entry.Type,
        entry.Brand, entry.Flavour, entry.Size_ml,
        entry.Caffeine_per_100, entry.Caffeine_total,
        entry.Feeling
      ]);
      sheet.getRange(2, 1, rows.length, 9).setValues(rows);
      sheet.getRange(2, 1, rows.length, 1).setNumberFormat('@STRING@');
      sheet.getRange(2, 2, rows.length, 1).setNumberFormat('@STRING@');
    }
    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService
    .createTextOutput(JSON.stringify({ ok: false, error: 'Unknown action' }))
    .setMimeType(ContentService.MimeType.JSON);
}
